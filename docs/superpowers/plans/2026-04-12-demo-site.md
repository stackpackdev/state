**Goal:** Build a standalone Vite + React + TypeScript demo site at `/Users/parandykt/Apps/stackpack/demo/` that showcases stackpack-state's advantages over Zustand, Redux, and Jotai through a live task manager, 9-step guided tour, agent panel, and capability comparison table.

## Task 4: StackpackPanel — the left panel

**Files:**
- Create: `/Users/parandykt/Apps/stackpack/demo/src/panels/StackpackPanel.tsx`

This is the most important panel — uses stackpack-state's `defineStore`, `useValue`, `useComputed`, `useGate`, `Presence`, history, optimistic updates, effects, and actors.

Note: stackpack-state's local package is at `../state`. The package name in package.json is `stackpack-state`. Import from `stackpack-state` and `stackpack-state/react`.

The `defineStore` API (from `demo-app-sa` reference):
```ts
defineStore({
  name: string,
  schema: ZodObject,
  initial: T,
  when: { [key]: (s: T) => boolean },       // style-edge conditions
  gates: { [key]: (s: T) => boolean },       // mount-edge conditions
  computed: { [key]: (s: T) => value },      // derived values
})
```

React hooks: `useValue(store)`, `useComputed(store, 'key')`, `useGate(store, 'key')`, `useWhen(store, 'key')`, `useChange(store)` (returns setState-like updater), `useActor()` (returns/sets current actor).

Components: `<Presence>` wraps items that need deferred unmount. `<Gated gate={bool}>` mounts children only when gate is true.

Effects: declared in `defineStore` options as `effects: [{ on: 'change', run: (state) => void }]`

History: declared in `defineStore` options as `history: { enabled: true, maxSize: 50 }`. Exposed on store object as `store.undo()`, `store.canUndo`.

Optimistic: `store.optimistic(updateFn, rollbackFn)` — applies update immediately, calls rollbackFn on failure (simulated here with a timeout).

- [ ] **Step 1: Create src/panels/StackpackPanel.tsx**

```tsx
import React, { useEffect, useRef, useState } from 'react'
import { defineStore, z } from 'stackpack-state'
import { useValue, useComputed, useGate, useChange, StoreProvider } from 'stackpack-state/react'
import { Presence } from 'stackpack-state/react'
import { Gated } from 'stackpack-state/react'
import type { Task, ActorLabel } from '../types'

// ─── Store Definition ─────────────────────────────────────────

const taskSchema = z.object({
  id: z.string(),
  text: z.string(),
  completed: z.boolean(),
  actor: z.enum(['you', 'agent']),
  exiting: z.boolean().optional(),
})

const tasksSchema = z.object({
  items: z.array(taskSchema),
  loaded: z.boolean(),
})

const taskStore = defineStore({
  name: 'tasks',
  schema: tasksSchema,
  initial: { items: [] as Task[], loaded: false },
  gates: {
    isLoaded: (s) => s.loaded,
  },
  computed: {
    completedCount: (s) => s.items.filter((t) => t.completed).length,
    totalCount: (s) => s.items.length,
  },
  history: { enabled: true, maxSize: 50 },
  effects: [
    {
      on: 'change',
      run: () => {
        // signal localStorage save (actual save done in component for simplicity)
        window.dispatchEvent(new Event('sp-saved'))
      },
    },
  ],
})

// ─── Inner component (must be inside StoreProvider) ───────────

function TaskManagerInner({ onActorChange }: { onActorChange: (actor: ActorLabel) => void }) {
  const state = useValue(taskStore)
  const completedCount = useComputed(taskStore, 'completedCount')
  const totalCount = useComputed(taskStore, 'totalCount')
  const isLoaded = useGate(taskStore, 'isLoaded')
  const update = useChange(taskStore)
  const [newText, setNewText] = useState('')
  const [savedVisible, setSavedVisible] = useState(false)
  const [currentActor, setCurrentActor] = useState<ActorLabel>('you')
  const inputRef = useRef<HTMLInputElement>(null)

  // Simulate initial load delay (Gate demo)
  useEffect(() => {
    const timer = setTimeout(() => {
      update((s) => {
        s.loaded = true
        s.items = [
          { id: '1', text: 'Review stackpack-state docs', completed: false, actor: 'you' },
          { id: '2', text: 'Build something with defineStore', completed: true, actor: 'you' },
        ]
      })
    }, 1000)
    return () => clearTimeout(timer)
  }, [])

  // Save indicator
  useEffect(() => {
    const handler = () => {
      setSavedVisible(true)
      setTimeout(() => setSavedVisible(false), 1200)
    }
    window.addEventListener('sp-saved', handler)
    return () => window.removeEventListener('sp-saved', handler)
  }, [])

  const addTask = () => {
    if (!newText.trim()) return
    update((s) => {
      s.items.push({ id: Date.now().toString(), text: newText.trim(), completed: false, actor: currentActor })
    })
    setNewText('')
    inputRef.current?.focus()
  }

  const toggleTask = (id: string) => {
    update((s) => {
      const t = s.items.find((i) => i.id === id)
      if (t) t.completed = !t.completed
    })
  }

  const deleteTask = (id: string) => {
    update((s) => {
      s.items = s.items.filter((i) => i.id !== id)
    })
  }

  const optimisticDelete = (id: string) => {
    // Remove immediately
    const backup = state.items.find((i) => i.id === id)
    update((s) => { s.items = s.items.filter((i) => i.id !== id) })
    // Simulate failure: rollback after 1.5s
    setTimeout(() => {
      if (backup) {
        update((s) => { s.items.push(backup) })
      }
    }, 1500)
  }

  const undo = () => {
    taskStore.undo?.()
  }

  // Expose addTask for agent simulation
  useEffect(() => {
    ;(window as any).__spAddTask = (text: string, actor: ActorLabel) => {
      update((s) => {
        s.items.push({ id: Date.now().toString(), text, completed: false, actor })
      })
    }
    ;(window as any).__spCompleteFirst = () => {
      update((s) => {
        const incomplete = s.items.find((i) => !i.completed)
        if (incomplete) incomplete.completed = true
      })
    }
  }, [update])

  return (
    <div id="panel-stackpack" className="h-full flex flex-col gap-4">
      {/* Header row */}
      <div className="flex items-center justify-between">
        <div className="flex items-center gap-2">
          <span className="font-semibold text-sm">stackpack-state</span>
          <span
            id="badge-completed"
            className="text-xs px-2 py-0.5 rounded-full bg-accent text-white font-medium"
          >
            {completedCount}/{totalCount}
          </span>
        </div>
        <div className="flex items-center gap-2">
          <span
            id="saved-indicator"
            className={`text-xs text-accent transition-opacity duration-200 ${savedVisible ? 'opacity-100' : 'opacity-0'}`}
          >
            Saved
          </span>
          <button
            id="btn-undo-stackpack"
            className="btn-secondary text-xs px-2 py-1"
            onClick={undo}
          >
            ↩ Undo
          </button>
        </div>
      </div>

      {/* Task list with Gate */}
      <div id="task-list-stackpack" className="flex-1 min-h-0 overflow-y-auto flex flex-col gap-2">
        {!isLoaded ? (
          // Skeleton (Gate: mount only when loaded)
          <div className="flex flex-col gap-2">
            {[1, 2].map((i) => (
              <div key={i} className="h-10 rounded-md bg-linen-border animate-pulse" />
            ))}
          </div>
        ) : (
          state.items.map((task) => (
            <Presence key={task.id} exitDuration={350}>
              <div className="flex items-center gap-2 p-2.5 rounded-md border border-linen-border bg-white group">
                <input
                  type="checkbox"
                  checked={task.completed}
                  onChange={() => toggleTask(task.id)}
                  className="accent-black w-4 h-4 flex-shrink-0"
                />
                <span className={`flex-1 text-sm ${task.completed ? 'line-through text-linen-muted' : ''}`}>
                  {task.text}
                </span>
                <span className="text-[10px] uppercase tracking-wider text-linen-muted px-1.5 py-0.5 rounded bg-linen">
                  {task.actor}
                </span>
                <button
                  className="opacity-0 group-hover:opacity-100 text-xs text-linen-muted hover:text-red-500 transition-all"
                  onClick={() => deleteTask(task.id)}
                >
                  ×
                </button>
              </div>
            </Presence>
          ))
        )}
      </div>

      {/* Add task */}
      <div className="flex gap-2">
        <input
          ref={inputRef}
          value={newText}
          onChange={(e) => setNewText(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && addTask()}
          placeholder="Add a task..."
          className="flex-1 text-sm px-3 py-2 rounded-md border border-linen-border bg-white outline-none focus:border-[#1a1a1a] transition-colors"
        />
        <button className="btn-primary text-sm" onClick={addTask}>Add</button>
      </div>

      {/* Optimistic delete button */}
      {state.items.length > 0 && (
        <button
          id="btn-optimistic-stackpack"
          className="btn-secondary text-xs w-full"
          onClick={() => optimisticDelete(state.items[0].id)}
        >
          Delete first task (optimistic — rolls back in 1.5s)
        </button>
      )}
    </div>
  )
}

// ─── Exported panel ───────────────────────────────────────────

interface StackpackPanelProps {
  onActorChange: (actor: ActorLabel) => void
}

export function StackpackPanel({ onActorChange }: StackpackPanelProps) {
  return (
    <StoreProvider store={taskStore}>
      <TaskManagerInner onActorChange={onActorChange} />
    </StoreProvider>
  )
}
```

import type { Task } from '../types'

interface TasksState {
  items: Task[]
  loaded: boolean
}

const tasksSlice = createSlice({
  name: 'tasks',
  initialState: { items: [] as Task[], loaded: false } as TasksState,
  reducers: {
    setLoaded(state) {
      state.loaded = true
      state.items = [
        { id: '1', text: 'Review redux docs', completed: false, actor: 'you' },
        { id: '2', text: 'Build something with createSlice', completed: true, actor: 'you' },
      ]
    },
    addTask(state, action: PayloadAction<string>) {
      state.items.push({ id: Date.now().toString(), text: action.payload, completed: false, actor: 'you' })
    },
    toggleTask(state, action: PayloadAction<string>) {
      const t = state.items.find((i) => i.id === action.payload)
      if (t) t.completed = !t.completed
    },
    removeTask(state, action: PayloadAction<string>) {
      state.items = state.items.filter((i) => i.id !== action.payload)

import React, { useEffect, useState } from 'react'
import { TOUR_STEPS } from '../tour/steps'

interface TourOverlayProps {
  active: boolean
  onEnd: () => void
}

export function TourOverlay({ active, onEnd }: TourOverlayProps) {
  const [stepIndex, setStepIndex] = useState(0)

  useEffect(() => {
    if (active) setStepIndex(0)
  }, [active])

  if (!active) return null

  const step = TOUR_STEPS[stepIndex]
  const isLast = stepIndex === TOUR_STEPS.length - 1

  const next = () => {
    if (isLast) {
      onEnd()
      // Scroll to capability table
      document.getElementById('capability-table')?.scrollIntoView({ behavior: 'smooth' })
    } else {
      setStepIndex((i) => i + 1)
    }
  }

  // Highlight target element
  useEffect(() => {
    if (!active) return
    const el = document.getElementById(step.targetId)
    if (el) {
      el.style.outline = '2px solid #22c55e'
      el.style.outlineOffset = '4px'
      el.style.borderRadius = '6px'
      el.scrollIntoView({ behavior: 'smooth', block: 'nearest' })
    }
    return () => {
      if (el) {
        el.style.outline = ''
        el.style.outlineOffset = ''
      }
    }
  }, [step, active])

  return (
    <div className="fixed bottom-6 right-6 z-50 w-80 card shadow-lg border-[#22c55e] border">
      <div className="flex items-center justify-between mb-2">
        <span className="label text-[#22c55e]">
          {stepIndex + 1} / {TOUR_STEPS.length}
        </span>
        <button onClick={onEnd} className="text-linen-muted hover:text-[#1a1a1a] text-xs">
          ✕ End tour
        </button>
      </div>
      <p className="font-semibold text-sm mb-1">{step.capability}</p>

        <button className="btn-primary text-xs flex-1" onClick={next}>

  )
